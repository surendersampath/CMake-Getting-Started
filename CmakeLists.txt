cmake_minimum_required(VERSION 3.12)
project( Get-Started-With-Cmake )

# Set C standard
set(CMAKE_C_STANDARD 11)

# Source files for the main application
set(MAIN_SOURCES
        src/main.c
)

# Create the test executable - This is to for unit testing our module.
add_subdirectory(unity)

# Create the test executable - This is to for unit testing our module.
add_subdirectory(test_module)

# Create the main application executable - You can consider this as running main.c
add_executable(MyProject ${MAIN_SOURCES})

# Link the main application with the test module library
target_link_libraries(MyProject Module)
