cmake_minimum_required(VERSION 3.12)
project( Get-Started-With-Cmake )

# Set C standard
set(CMAKE_C_STANDARD 11)

# Source files for the main application
set(MAIN_SOURCES
        src/main.c
)

# Include Subdirectoy for test-framework.
add_subdirectory(unity)

# Include Subdirectoy for including the module library.
add_subdirectory(test_module)

# Include directories for module library and its dependencies
include_directories(
        test_module/include
)

# Create the main application executable - You can consider this as running main.c
add_executable(MyProject ${MAIN_SOURCES})

# Link the main application with the module library - To Access Functions inside the module.
target_link_libraries(MyProject Module)
